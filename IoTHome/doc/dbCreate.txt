#Oczekiwana temperatura powietrza w budynku
CREATE TABLE expected_temperature (time bigint PRIMARY KEY, T real);
#ustawiona temperatura na boilerze
CREATE TABLE boiler_set_temperature (time bigint PRIMARY KEY, T real);
#Pomiary przez termometry temperatur i gaz lub dym
CREATE TABLE measure (th varchar(50), time bigint, T real, h real, p real, s real);
#Dane ze stanu boilera
CREATE TABLE boiler_info (time bigint PRIMARY KEY, return_temperature real, boiler_temperature real, 
	setpoint_bound real, oem_diagnostic int);
CREATE TABLE camera_images_info (create_time bigint, camera_name varchar(100), PRIMARY KEY(create_time, camera_name));
CREATE TABLE pzem_info (create_time bigint, measurer varchar(100), v float, c float, p float, e float, f float, pf float, PRIMARY KEY(create_time, measurer));
CREATE TABLE watering_info (watering_time bigint, device varchar(100), PRIMARY KEY(watering_time, device));
CREATE TABLE moisture_info (moisture_time bigint, device varchar(100), moisture integer, PRIMARY KEY(moisture_time, device));
CREATE TABLE christmas_tree (id int, state int, PRIMARY KEY(id));
